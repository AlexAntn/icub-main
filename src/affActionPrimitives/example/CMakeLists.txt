SET(PROJECTNAME affActionPrimitivesExample)

PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP)
FIND_PACKAGE(ICUB)
FIND_PACKAGE(AFFACTIONPRIMITIVES REQUIRED)

IF(USE_ICUB_MOD)
    ADD_DEFINITIONS(-DUSE_ICUB_MOD)
	FIND_PACKAGE(icubmod)
	IMPORT_DEVICES(${AFFACTIONPRIMITIVES_DIR}/example/drivers.h icubmod)
	SET(folder_header drivers.h)
ELSE(USE_ICUB_MOD)
    SET(folder_header)
ENDIF(USE_ICUB_MOD)

SET(folder_source main.cpp)

SOURCE_GROUP("Source Files" FILES ${folder_source})
SOURCE_GROUP("Header Files" FILES ${folder_header})

INCLUDE_DIRECTORIES(${AFFACTIONPRIMITIVES_INCLUDE_DIRS})
ADD_EXECUTABLE(${PROJECTNAME} ${folder_source} ${folder_header})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${AFFACTIONPRIMITIVES_LIBRARIES})
