SET(PROJECTNAME randArmGazeMotion)
PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP)
FIND_PACKAGE(ICUB)
FIND_PACKAGE(OpenCV REQUIRED)

IF(NOT OPENCV_FOUND)
	MESSAGE(FATAL_ERROR "OpenCV is striclty required!")
ENDIF(NOT OPENCV_FOUND)

INCLUDE_DIRECTORIES(${OPENCV_INCLUDE_DIR})

FIND_PACKAGE(CTRLLIB REQUIRED)
INCLUDE_DIRECTORIES(${CTRLLIB_INCLUDE_DIRS})

SET(folder_source main.cpp)

SOURCE_GROUP("Source Files" FILES ${folder_source})


ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})


# Add CTRLLIB dependencies
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${CTRLLIB_LIBRARIES})

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${OPENCV_LIBRARIES})



IF(MSVC)
   SET_TARGET_PROPERTIES(${PROJECTNAME} PROPERTIES COMPILE_FLAGS /wd4996)
ENDIF(MSVC)
