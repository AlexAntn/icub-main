SET(PROJECTNAME zeroForceControl)

PROJECT(${PROJECTNAME})

SET(YARP_DIR "$ENV{YARP_ROOT}" CACHE LOCATION "where is the yarp build directory?")
SET(ICUB_DIR "$ENV{ICUB_ROOT}" CACHE LOCATION "where is icub root?")

FIND_PACKAGE(ICUB)
FIND_PACKAGE(IKIN REQUIRED)
INCLUDE_DIRECTORIES(${IKIN_INCLUDE_DIRS})
INCLUDE_DIRECTORIES(${IFC_INCLUDE_DIRS}})

FIND_PACKAGE(IFC REQUIRED)
INCLUDE_DIRECTORIES(${IFC_INCLUDE_DIRS})

SET(folder_source main.cpp)
SOURCE_GROUP("Source Files" FILES ${folder_source})

ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${IKIN_LIBRARIES})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${IFC_LIBRARIES})

INSTALL(TARGETS zeroForceControl RUNTIME DESTINATION bin)
