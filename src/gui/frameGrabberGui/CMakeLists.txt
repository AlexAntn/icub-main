PROJECT(frameGrabberGui)

FIND_PACKAGE(ICUB)
FIND_PACKAGE(YARP REQUIRED)

IF (NOT GtkMM_FOUND)
	FIND_PACKAGE(GtkMM REQUIRED)
ENDIF(NOT GtkMM_FOUND)

IF (GtkMM_FOUND)
	ADD_DEFINITIONS(${GTKMM_C_FLAGS})
	LINK_LIBRARIES(${GTKMM_LINK_FLAGS})
	INCLUDE_DIRECTORIES(${GTKMM_INCLUDE_DIR})

	ADD_EXECUTABLE(frameGrabberGui fggui_main.cpp  FrameGrabberGUIControl.h)

	IF (WIN32 AND NOT CYGWIN)
  		SET_TARGET_PROPERTIES(frameGrabberGui PROPERTIES COMPILE_FLAGS /wd4250)
	ENDIF (WIN32 AND NOT CYGWIN)

	INSTALL_TARGETS(/bin frameGrabberGui)
ELSEIF(GtkMM_FOUND)
	MESSAGE("GTKMM not found, won't compile frameGrabberGui")
ENDIF(GtkMM_FOUND)
