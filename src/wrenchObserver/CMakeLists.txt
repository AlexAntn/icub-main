SET(PROJECTNAME wrenchObserver)

MESSAGE(STATUS "--> adding wrenchObserver")

PROJECT(${PROJECTNAME})

# YARP needs CMake 2.6 or greater
cmake_minimum_required(VERSION 2.6)

# find YARP
find_package(YARP REQUIRED)

# add YARP include directories
include_directories(${YARP_INCLUDE_DIRS})

FIND_PACKAGE(ICUB)
include_directories(${GSL_INCLUDE_DIRS})

FIND_PACKAGE(IKIN REQUIRED)
INCLUDE_DIRECTORIES(${IKIN_INCLUDE_DIRS})
FIND_PACKAGE(IDYN REQUIRED)
INCLUDE_DIRECTORIES(${IDYN_INCLUDE_DIRS})

SET(folder_source main.cpp)
SOURCE_GROUP("Source Files" FILES ${folder_source})

ADD_EXECUTABLE(objectDetector ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${YARP_LIBRARIES} ${IDYN_LIBRARIES} ${IKIN_LIBRARIES})
MESSAGE(STATUS ${IDYN_LIBRARIES})
MESSAGE(STATUS ${GSL_INCLUDE_DIRS})


INSTALL(TARGETS ${PROJECTNAME} DESTINATION bin) #RUNTIME DESTINATION bin
