PROJECT(goFlow)

FIND_PACKAGE(YARP)
FIND_PACKAGE(OpenCV)

SET(PROJECT_DIR ${CMAKE_SOURCE_DIR})
SET(HEADER_PATH "${PROJECT_DIR}/include")
SET(SRC_PATH "${PROJECT_DIR}/src")

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR} ${OPENCV_INCLUDE_DIR} ${HEADER_PATH})

LINK_DIRECTORIES(${OPENCV_LINK_DIRECTORIES})
LINK_LIBRARIES(${OPENCV_LIBRARIES})

SET(CODEFILES src/GoFlow.cpp src/Flow.cpp src/FlowJob.cpp)

SOURCE_GROUP("Header Files" FILES ${HEADERFILES})
SOURCE_GROUP("Source Files" FILES ${CODEFILES})

ADD_EXECUTABLE(goFlow  ${CODEFILES})
ADD_EXECUTABLE(fileImgSrc  src/FileImageSrc.cpp)
ADD_EXECUTABLE(goFlowView src/GoFlowView.cpp)
