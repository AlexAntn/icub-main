cmake_minimum_required(VERSION 2.6)

SET(PROJECTNAME lmExample)

SET(LMLIB_DIRECT lmlibdirect)
SET(LMLIB_INDIRECT lmlibindirect)
SET(LMLIB_PORTABLE lmlibportable)

PROJECT(${PROJECTNAME})
FIND_PACKAGE(YARP)
FIND_PACKAGE(ICUB QUIET)
IF (NOT ICUB_FOUND)
   MESSAGE (STATUS "ICUB repository not found; assuming distributed build.")
   MESSAGE (STATUS "If this is not what you want, set ICUB_DIR variable.")
ENDIF (NOT ICUB_FOUND)

SET(LEARNINGMACHINE_DIR "${ICUB_DIR}/src/learningMachine")

FIND_PACKAGE(LEARNINGMACHINE REQUIRED)

LINK_LIBRARIES(${LEARNINGMACHINE_LIBRARIES})
INCLUDE_DIRECTORIES(${LEARNINGMACHINE_INCLUDE_DIRS})

ADD_EXECUTABLE(${LMLIB_DIRECT} ${LMLIB_DIRECT}.cpp)
ADD_EXECUTABLE(${LMLIB_INDIRECT} ${LMLIB_INDIRECT}.cpp)
