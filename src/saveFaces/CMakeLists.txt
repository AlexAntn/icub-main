cmake_minimum_required(VERSION 2.6)

SET(PROJECTNAME saveFaces)

PROJECT(${PROJECTNAME})


FIND_PACKAGE(Qt3 REQUIRED)
INCLUDE_DIRECTORIES(${QT3_INCLUDE_DIR})
LINK_LIBRARIES(${QT3_LIBRARIES})
ADD_DEFINITIONS(${QT_DEFINITIONS})
ADD_DEFINITIONS(-DQT_THREAD_SUPPORT)

FIND_PACKAGE(YARP)
FIND_PACKAGE(ICUB)
FIND_PACKAGE(OpenCV)

IF(OPENCV_FOUND)
   ADD_DEFINITIONS(-DADD_VIDEO)
   INCLUDE_DIRECTORIES(${OPENCV_INCLUDE_DIR})
ELSE(OPENCV_FOUND)
   MESSAGE("${PROJECTNAME}: OpenCV not found! Keep on building...")
ENDIF(OPENCV_FOUND)

IF(YARP_FOUND)
ELSE(YARP_FOUND)
   MESSAGE("${PROJECTNAME}: YARP not found! Keep on building...")
ENDIF(YARP_FOUND)



SET(folder_source torsodet.cpp saveFaces.cpp   )
SET(folder_header torsodet.h )

SOURCE_GROUP("Source Files" FILES ${folder_source})

ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})
IF(OPENCV_FOUND)
   TARGET_LINK_LIBRARIES(${PROJECTNAME} ${OPENCV_LIBRARIES})
ENDIF(OPENCV_FOUND)
INSTALL(TARGETS ${PROJECTNAME} DESTINATION bin)
