SET(PROJECTNAME iKinGazeView)

PROJECT(${PROJECTNAME})

FIND_PATH(MATLAB_INCLUDE_DIRS engine.h NO_DEFAULT_PATH)
IF(WIN32)
   FIND_PATH(MATLAB_LIBRARIES_DIRS libeng.lib NO_DEFAULT_PATH)
ELSE(WIN32)
   FIND_PATH(MATLAB_LIBRARIES_DIRS eng NO_DEFAULT_PATH)
ENDIF(WIN32)

IF(WIN32)
   SET(MATLAB_LIBRARIES ${MATLAB_LIBRARIES_DIRS}/libeng.lib ${MATLAB_LIBRARIES_DIRS}/libmx.lib ${MATLAB_LIBRARIES_DIRS}/libmex.lib ${MATLAB_LIBRARIES_DIRS}/libmat.lib)
ELSE(WIN32)
   SET(MATLAB_LIBRARIES ${MATLAB_LIBRARIES_DIRS}/libeng.so ${MATLAB_LIBRARIES_DIRS}/libmx.so ${MATLAB_LIBRARIES_DIRS}/libmex.so ${MATLAB_LIBRARIES_DIRS}/libmat.so)
ENDIF(WIN32)

SET(folder_source main.cpp)

SOURCE_GROUP("Source Files" FILES ${folder_source})

INCLUDE_DIRECTORIES(${ctrlLib_INCLUDE_DIRS}
                    ${MATLAB_INCLUDE_DIRS}
                    ${YARP_INCLUDE_DIRS})

ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})

TARGET_LINK_LIBRARIES(${PROJECTNAME} ctrlLib
                                     ${MATLAB_LIBRARIES}
                                     ${YARP_LIBRARIES)

INSTALL(TARGETS ${PROJECTNAME} DESTINATION bin)
