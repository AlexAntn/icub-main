SET(PROJECTNAME tactileGrasp)

#MESSAGE(STATUS "--> adding tactileGrasp")

PROJECT(${PROJECTNAME})

FIND_PACKAGE(ICUB)

FIND_PACKAGE(IFC REQUIRED)
INCLUDE_DIRECTORIES(${IFC_INCLUDE_DIRS}  ${IKIN_INCLUDE_DIRS})

SET(folder_source main.cpp)
SOURCE_GROUP("Source Files" FILES ${folder_source})

ADD_EXECUTABLE(tactileGrasp ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${IFC_LIBRARIES} ${IKIN_LIBRARIES})
MESSAGE(STATUS ${IFC_LIBRARIES})


INSTALL(TARGETS ${PROJECTNAME} RUNTIME DESTINATION bin)
