SET(KEYWORD "yarl")

PROJECT(${KEYWORD})

FIND_PACKAGE(YARP REQUIRED)
FIND_LIBRARY(KDL_LIB NAMES kdl libkdl PATHS "$ENV{KDL_DIR}/kdl")
INCLUDE_DIRECTORIES(${PROJECT_SOURCE_DIR} $ENV{KDL_DIR})

# Set up our main executable.
ADD_EXECUTABLE(${KEYWORD} yarl.cpp)

# Add libraries
ADD_LIBRARY(${KEYWORD} ${KDL_LIB})
TARGET_LINK_LIBRARIES(${KEYWORD} ${KDL_LIB})
