PROJECT(roboop)
IF(UNIX)
  FIND_PACKAGE(Boost)
  INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIRS})
ENDIF(UNIX)
IF(WIN32)
  IF(NOT UNIX)
    ADD_DEFINITIONS(-D_STANDARD_)
  ENDIF(NOT UNIX)
ENDIF(WIN32)
SUBDIRS(source newmat)
INCLUDE_DIRECTORIES(source newmat)
SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR})
SET(LIBRARY_OUTPUT_PATH ${PROJECT_BINARY_DIR})



#By amaldo:
#SET(CMAKE_INSTALL_PREFIX /tmp/roboop)
#To have a good CMAKE configuration file (for FIND_PACKAGE):
SET(ROBOOP_LIBRARIES "roboop" "newmat")
SET(ROBOOP_INCLUDES "${CMAKE_CURRENT_SOURCE_DIR}/source" "${CMAKE_CURRENT_SOURCE_DIR}/newmat")
SET(ROBOOP_LINK_DIRECTORIES "${CMAKE_BINARY_DIR}")
CONFIGURE_FILE(ROBOOPConfig.cmake.in ${CMAKE_CURRENT_SOURCE_DIR}/ROBOOPConfig.cmake @ONLY IMMEDIATE)
