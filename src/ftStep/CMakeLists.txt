SET(PROJECTNAME ftStep)

MESSAGE(STATUS "--> adding ftStep")

PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP)
FIND_PACKAGE(ICUB)

FIND_PACKAGE(IKIN REQUIRED)
INCLUDE_DIRECTORIES(${IKIN_INCLUDE_DIRS})

FIND_PACKAGE(IFC REQUIRED)
INCLUDE_DIRECTORIES(${IFC_INCLUDE_DIRS})

SET(folder_source main.cpp filter.cpp)
SET(folder_header filter.h)
SOURCE_GROUP("Source Files" FILES ${folder_source} ${folder_header})

ADD_EXECUTABLE(ftStep ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${IKIN_LIBRARIES})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${IFC_LIBRARIES})

INSTALL(TARGETS ${PROJECTNAME} DESTINATION bin)
